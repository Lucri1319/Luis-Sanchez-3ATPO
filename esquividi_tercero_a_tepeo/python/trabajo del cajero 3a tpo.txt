balance = 1000
pin_intentos = 0

while True:
    pin = int(input(" Ingresa tu pin: "))
    if pin == 1234:
        while True:
            print(" --- Selecione una opcion del menu --- ")
            print(" 1. Consultar su saldo ")
            print(" 2. Retiro")
            print(" 3. Deposito ")
            print(" 4. Salir ")
            opcion = int(input(" Ingresa una opcion: "))
            
            if opcion == 1:
                print("Su saldo es de: ", balance)
                
            elif opcion == 2:   
                retirar = int(input(" Ingrese lo que desee retirar: "))
                if retirar > balance:
                    print(" Saldo insuficiente ")
                else:
                    balance -= retirar
                    print(" Has retirado: ", retirar)
                    print(" Tu nuevo saldo es: ", balance) 
            
            elif opcion == 3:
                depositar = int(input(" Ingresa la cantidad a depositar: "))
                balance += depositar
                print(" Has depositado: ", depositar)
                print(" Tu nuevo saldo es: ", balance)
            
            elif opcion == 4:
                print(" ¡Gracias! ")
                break
    else :
        print(" Pin incorrecto ")
        pin_intentos += 1
        if pin_intentos == 3:
            print(" Inténtelo más tarde ")
            break
